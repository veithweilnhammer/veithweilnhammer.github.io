import numpy as np
import matplotlib.pyplot as plt

# Function to generate responses using von Mises distribution with circular operations
def generate_response_circular(stimulus, previous_response, weights, bias, kappa):
    # Combine the circular variables using sine and cosine
    combined_sin = weights[0] * np.sin(stimulus) + weights[1] * np.sin(bias) + weights[2] * np.sin(previous_response)
    combined_cos = weights[0] * np.cos(stimulus) + weights[1] * np.cos(bias) + weights[2] * np.cos(previous_response)
    
    # Calculate the mean orientation using atan2 to handle circular space
    mean_orientation = np.arctan2(combined_sin, combined_cos)
    
    # Generate a response based on von Mises distribution centered on the mean orientation
    response = np.random.vonmises(mean_orientation, kappa)
    
    return response

def process_mat_file(file_path, subject_idx, session_idx, location):
    # Load the MAT file
    mat_data = scipy.io.loadmat(file_path)

    # Extract relevant data
    target_orientations = mat_data['TargetOrientations'].flatten()
    response_orientations = mat_data['ResponseOrientations'].flatten()

    # Convert from degrees to radians (pi units)
    target_orientations = np.deg2rad(target_orientations)  # Convert to radians
    response_orientations = np.deg2rad(response_orientations)  # Convert to radians

    # Create a DataFrame
    df = pd.DataFrame({
        'TargetOrientation': target_orientations,
        'ResponseOrientation': response_orientations
    })

    # Add PreviousResponseOrientation (shift ResponseOrientation by one)
    df['PreviousResponseOrientation'] = df['ResponseOrientation'].shift(1)
    df['PreviousResponseOrientation'].fillna(0, inplace=True)

    # Rename the columns to match the expected format
    df.rename(columns={
        'TargetOrientation': 'stimuli',
        'ResponseOrientation': 'responses',
        'PreviousResponseOrientation': 'previous_responses'
    }, inplace=True)

    # Add subject, session, and location information
    df['subject'] = subject_idx
    df['session'] = session_idx
    df['location'] = location

    return df

import numpy as np
import ssm
import matplotlib.pyplot as plt
from itertools import cycle
from sklearn.preprocessing import StandardScaler

# Function to fit HMM and return results (separate estimation)
def fit_hmm(n_states, all_data, bias):
    obs_dim = 1  # Single response (von Mises)
    input_dim = 1  # Stimulus, bias, and previous response
    n_trials = len(all_data)  # Number of trials

    # Prepare the input matrix for the HMM
    inpts = np.ones((n_trials, input_dim))
    inpts[:, 0] = all_data['stimuli'] - all_data['previous_responses']  # Previous responses

    inpts = inpts

    # Prepare the binary choices for HMM
    choices = np.atleast_2d((all_data['stimuli'] > all_data['responses']).astype(int)).T

    # Fit the HMM with the given number of states
    hmm = ssm.HMM(n_states, obs_dim, input_dim, 
                  observations="input_driven_obs", 
                  observation_kwargs=dict(C=2, prior_sigma=10), 
                  transitions="sticky", 
                  transition_kwargs=dict(alpha=1, kappa=0))

    # Fit the HMM using Expectation-Maximization (EM)
    hmm.fit(choices, inputs=inpts, method="em", num_iters=500, initialize=True, tolerance=10**-4, verbose=False)

    # Get the most likely states from the fitted HMM
    inferred_states = hmm.most_likely_states(choices, input=inpts)

    # Compute BIC
    ll = hmm.log_likelihood(choices, inputs=inpts)
    n_parameters = np.concatenate(hmm.observations.params).size + np.concatenate(hmm.transitions.params).size
    bic = np.log(n_trials) * n_parameters - 2 * ll

    # Get expected posterior probabilities
    posterior_probs = hmm.expected_states(data=choices, input=inpts)[0]

    # Extract inferred weights (parameters of the input-driven observations)
    inferred_weights = hmm.observations.params

    return hmm, inferred_states, posterior_probs, bic, inferred_weights

# Function to plot results (separate plotting)
def plot_hmm_results(n_states, all_data, inferred_states, posterior_probs, bic, inferred_weights, num_plot_trials=1500):
    # Define a color cycle for plotting inferred states
    colors_inferred = list(plt.cm.tab10.colors[:n_states])  # Get colors for the states

    # Calculate the circular differences for plotting
    response_stimulus_diff = (all_data['responses'] - all_data['stimuli'] + np.pi) % (2 * np.pi) - np.pi  # Circular difference
    stimulus_diff = (all_data['stimuli'] - all_data['previous_responses'] + np.pi) % (2 * np.pi) - np.pi  # Circular difference

    # Bin the stimulus differences
    bins = np.linspace(-np.pi, np.pi, 200)
    digitized = np.digitize(stimulus_diff, bins)
    
    # Plot 1: Average response-stimulus difference
    plt.figure(figsize=(8, 6))
    for state_idx in range(n_states):
        bin_means_inferred = [response_stimulus_diff[(digitized == i) & (inferred_states == state_idx)].mean() for i in range(1, len(bins))]
        plt.plot(bins[:-1], bin_means_inferred, linestyle='-', color=colors_inferred[state_idx], label=f'Inferred State {state_idx + 1}')

    # Customize the plot for inferred states
    plt.title("Average Response-Stimulus Difference (Inferred States)")
    plt.xlabel("Stimulus Difference (Radians) [Stimulus(t) - Response(t-1)]")
    plt.ylabel("Average Response-Stimulus Difference (Radians)")
    plt.grid(True)
    plt.legend()
    plt.tight_layout()
    plt.show()

    # Plot 2: Posterior probabilities for a session (choose a session ID)
    fig = plt.figure(figsize=(5, 2.5), dpi=80, facecolor='w', edgecolor='k')
    for k in range(n_states):
        plt.plot(posterior_probs[1:num_plot_trials, k], label="State " + str(k + 1), lw=2, color=colors_inferred[k])
    plt.ylim((-0.01, 1.01))
    plt.yticks([0, 0.5, 1], fontsize=10)
    plt.xlabel("Trial #", fontsize=15)
    plt.ylabel("p(State)", fontsize=15)
    plt.legend()
    plt.tight_layout()
    plt.show()

    # Get state with maximum posterior probability at each trial
    state_max_posterior = np.argmax(posterior_probs, axis=1)

    # Compute state fractional occupancies
    _, state_occupancies = np.unique(state_max_posterior, return_counts=True)
    state_occupancies = state_occupancies / np.sum(state_occupancies)

    # Plot 3: State fractional occupancies with same colors as above
    fig = plt.figure(figsize=(4, 3), dpi=80, facecolor='w', edgecolor='k')
    for z, occ in enumerate(state_occupancies):
        plt.bar(z, occ, width=0.8, color=colors_inferred[z])
    plt.ylim((0, 1))
    plt.xticks(range(n_states), [str(i + 1) for i in range(n_states)], fontsize=10)
    plt.yticks([0, 0.5, 1], ['0', '0.5', '1'], fontsize=10)
    plt.xlabel('State', fontsize=15)
    plt.ylabel('Fractional Occupancy', fontsize=15)
    plt.title("State Occupancy")
    plt.tight_layout()
    plt.show()

    print(f"BIC for {n_states} states: {bic}")
    print("Inferred Weights:")
    for i, w in enumerate(inferred_weights):
        print(f"State {i + 1}: {w}")
